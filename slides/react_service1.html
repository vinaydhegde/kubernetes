React Service


k8s/devopstree-service-react.yml:


apiVersion: v1
kind: Service
metadata:
  name: devopstree-react
  labels:
    service: devopstree-react
  name: devopstree-react
spec:
  selector:
    app: devopstree-react
  ports:
  - port: 8080
    targetPort: 8080

Create the Service


$ kubectl create -f ./k8s/devopstree-service-react.yml
