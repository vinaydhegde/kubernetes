Postgres Service


k8s/devopstree-service-postgres.yml:


apiVersion: v1
kind: Service
metadata:
  name: devopstree-postgres
  labels:
    service: devopstree-postgres
spec:
  selector:
    service: devopstree-postgres
  type: ClusterIP
  ports:
  - port: 5432

Create the Service:


$ kubectl create -f ./k8s/devopstree-service-postgres.yml
